```python
class GraphStore:
    """基于 NetworkX 的知识图谱存储."""

    def __init__(self):
        self.graph = nx.DiGraph()
        self.tag_index = {}  # 标签索引

    def add_memory(
        self,
        content: str,
        tags: list[str],
        metadata: dict
    ) -> str:
        """添加记忆节点."""
        node_id = self._generate_id()
        self.graph.add_node(
            node_id,
            content=content,
            tags=tags,
            metadata=metadata,
            timestamp=time.time()
        )
        self._update_tag_index(node_id, tags)
        return node_id

    def search_memory(
        self,
        query: str,
        tags: list[str],
        limit: int
    ) -> list[dict]:
        """搜索记忆."""
        # 1. 标签过滤
        candidates = self._filter_by_tags(tags)

        # 2. 内容匹配
        matches = self._match_content(candidates, query)

        # 3. 相关性排序
        sorted_matches = self._rank_by_relevance(matches)

        return sorted_matches[:limit]
```

## 关键算法

### 1. 相关性排序

```python
def _rank_by_relevance(matches: list[dict], query: str) -> list[dict]:
    """基于多因素的相关性排序."""

    def score(memory: dict) -> float:
        # 1. 内容匹配度（TF-IDF）
        content_score = tfidf_score(memory["content"], query)

        # 2. 标签匹配度
        tag_score = len(set(memory["tags"]) & set(query_tags))

        # 3. 时间衰减
        time_score = time_decay(memory["timestamp"])

        # 4. 重要性权重
        importance = memory["metadata"].get("importance", "medium")
        importance_score = {"low": 0.5, "medium": 1.0, "high": 1.5}[importance]

        # 综合评分
        return (
            content_score * 0.4 +
            tag_score * 0.3 +
            time_score * 0.2 +
            importance_score * 0.1
        )

    return sorted(matches, key=score, reverse=True)
```

### 2. 图遍历查询

```python
def find_related_memories(
    node_id: str,
    max_depth: int = 2
) -> list[str]:
    """查找相关记忆（广度优先遍历）."""

    visited = set()
    queue = [(node_id, 0)]  # (node, depth)
    related = []

    while queue:
        current, depth = queue.pop(0)

        if current in visited or depth > max_depth:
            continue

        visited.add(current)
        related.append(current)

        # 添加邻居节点
        for neighbor in graph.neighbors(current):
            if neighbor not in visited:
                queue.append((neighbor, depth + 1))

    return related
```

### 3. 记忆合并去重

```python
def merge_similar_memories(threshold: float = 0.85) -> list[tuple]:
    """合并相似记忆."""

    memories = list(graph.nodes(data=True))
    merged_pairs = []

    for i, (id1, data1) in enumerate(memories):
        for id2, data2 in memories[i+1:]:
            similarity = cosine_similarity(
                data1["embedding"],
                data2["embedding"]
            )

            if similarity > threshold:
                # 合并节点
                merged_content = merge_content(
                    data1["content"],
                    data2["content"]
                )
                merged_pairs.append((id1, id2, merged_content))

    return merged_pairs
```
